import pandas as pd
import matplotlib.pyplot as plt

def analyze_patient_data(data):
    # Perform statistical analysis
    summary = data.describe()
    
    # Create visualizations
    plt.figure(figsize=(10, 6))
    data['blood_pressure'].plot(kind='hist')
    plt.title('Distribution of Blood Pressure Readings')
    plt.xlabel('Blood Pressure')
    plt.ylabel('Frequency')
    plt.savefig('blood_pressure_distribution.png')
    
    return summary

# Usage
patient_data = pd.read_csv('patient_records.csv')
results = analyze_patient_data(patient_data)
print(results)

Sample Artifact: Fraud Detection Algorithm

import numpy as np
from sklearn.ensemble import IsolationForest

def detect_fraud(transactions):
    # Prepare features for anomaly detection
    features = transactions[['amount', 'time_since_last_transaction', 'distance_from_home']]
    
    # Train Isolation Forest model
    model = IsolationForest(contamination=0.05, random_state=42)
    predictions = model.fit_predict(features)
    
    # Flag potential fraudulent transactions
    transactions['is_fraud'] = np.where(predictions == -1, True, False)
    
    return transactions[transactions['is_fraud']]

# Usage
transaction_data = pd.read_csv('transactions.csv')
suspicious_transactions = detect_fraud(transaction_data)
print(suspicious_transactions)

Sample Artifact: Personalized Learning Path Generator

from sklearn.cluster import KMeans
import numpy as np

def generate_learning_path(student_data):
    # Extract relevant features
    features = student_data[['quiz_scores', 'time_spent', 'preferred_learning_style']]
    
    # Cluster students into groups
    kmeans = KMeans(n_clusters=3, random_state=42)
    clusters = kmeans.fit_predict(features)
    
    # Generate personalized paths based on cluster
    paths = []
    for cluster in clusters:
        if cluster == 0:
            paths.append("Visual-heavy content with frequent quizzes")
        elif cluster == 1:
            paths.append("Text-based content with in-depth projects")
        else:
            paths.append("Interactive simulations with peer collaboration")
    
    student_data['recommended_path'] = paths
    return student_data

# Usage
student_data = pd.read_csv('student_profiles.csv')
personalized_paths = generate_learning_path(student_data)
print(personalized_paths)
